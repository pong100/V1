local function createPart(p1, p2, fixedY, thicknessY)
	local size = Vector3.new(
		math.abs(p2.X - p1.X),
		thicknessY,
		math.abs(p2.Z - p1.Z)
	)

	local center = Vector3.new(
		(p1.X + p2.X) / 2,
		fixedY,
		(p1.Z + p2.Z) / 2
	)

	local part = Instance.new("Part")
	part.Size = size
	part.Position = center
	part.Anchored = true
	part.Material = Enum.Material.SmoothPlastic
	part.Color = color
	part.Parent = folder
end

-- =========================================
-- FUNCTION CHIA SEGMENT
-- =========================================
local function createSegments(p1, p2, fixedY, thicknessY)
	local minX = math.min(p1.X, p2.X)
	local maxX = math.max(p1.X, p2.X)

	local minZ = math.min(p1.Z, p2.Z)
	local maxZ = math.max(p1.Z, p2.Z)

	local depth = maxZ - minZ
	local centerZ = (minZ + maxZ) / 2

	local currentX = minX

	while currentX < maxX do
		local nextX = math.min(currentX + segmentLength, maxX)
		local width = nextX - currentX

		local part = Instance.new("Part")
		part.Size = Vector3.new(width, thicknessY, depth)
		part.Position = Vector3.new(
			currentX + width / 2,
			fixedY,
			centerZ
		)
		part.Anchored = true
		part.Material = Enum.Material.SmoothPlastic
		part.Color = color
		part.Parent = folder

		currentX = nextX
	end
end

-- =============================
-- FLY FUNCTION
-- =============================
local function flyToTarget(targetPosition)

if isFlying or cooldown then  
    return  
end  

isFlying = true  
cooldown = true  

local character = player.Character or player.CharacterAdded:Wait()  
local hrp = character:WaitForChild("HumanoidRootPart")  
local humanoid = character:WaitForChild("Humanoid")  

-- Noclip  
local noclip  
noclip = RunService.Stepped:Connect(function()  
    for _, v in pairs(character:GetDescendants()) do  
        if v:IsA("BasePart") then  
            v.CanCollide = false  
        end  
    end  
end)  

-- BodyVelocity  
local bodyVelocity = Instance.new("BodyVelocity")  
bodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)  
bodyVelocity.Parent = hrp  

-- Adjust Y first  
local adjustPos = Vector3.new(hrp.Position.X, targetY, hrp.Position.Z)  
hrp.CFrame = CFrame.new(adjustPos)  

repeat  
    local direction = (targetPosition - hrp.Position)  
    local distance = direction.Magnitude  

    local speed = flySpeed  
    if distance < 50 then  
        speed = 150  
    end  

    bodyVelocity.Velocity = direction.Unit * speed  
    task.wait()  
until (hrp.Position - targetPosition).Magnitude < 3  

bodyVelocity:Destroy()  
hrp.CFrame = CFrame.new(targetPosition)  

if noclip then  
    noclip:Disconnect()  
end  

for _, v in pairs(character:GetDescendants()) do  
    if v:IsA("BasePart") then  
        v.CanCollide = true  
    end  
end  

isFlying = false  

task.delay(3, function()  
    cooldown = false  
end)

    Title = "Teleport Complete",  
    Text = "Đã tới: "..selectedRarity,  
    Duration = 3

local FINAL_POSITION = Vector3.new(128.63, 3.26, 17.39)

-- ===== STATE =====
local selectedTier = "Common"

-- =============================
-- CORE FARM
-- =============================

local function startTierFarm()

    -- nếu còn thread cũ thì huỷ
    farmThread = nil

    farmThread = task.spawn(function()

        while tierEnabled do

            local character = player.Character
            if not character then task.wait() continue end

            local root = character:WaitForChild("HumanoidRootPart", 2)
            if not root then task.wait() continue end

            local folder = workspace:FindFirstChild("ActiveBrainrots")
            if not folder then task.wait(1) continue end

            local tierFolder = folder:FindFirstChild(selectedTier)
            if not tierFolder or #tierFolder:GetChildren() == 0 then
                task.wait(1)
                continue
            end

            -- ===== NOCLIP =====
            if not noclipConnection then
                noclipConnection = RunService.Stepped:Connect(function()
                    if tierEnabled and player.Character then
                        for _, v in pairs(player.Character:GetDescendants()) do
                            if v:IsA("BasePart") then
                                v.CanCollide = false
                            end
                        end
                    end
                end)
            end

            -- ===== BODY MOVERS =====
            local bv = Instance.new("BodyVelocity")
            bv.MaxForce = Vector3.new(1e9,1e9,1e9)
            bv.Parent = root

            local bp = Instance.new("BodyPosition")
            bp.MaxForce = Vector3.new(1e9,1e9,1e9)
            bp.P = 10000
            bp.D = 1000
            bp.Parent = root

            local function flyTo(position)
                bp.Position = position

                while tierEnabled
                and player.Character
                and root.Parent
                and (root.Position - position).Magnitude > 3 do

                    local distance = (root.Position - position).Magnitude
                    local dir = (position - root.Position).Unit
                    local speed = distance < 40 and 300 or FLY_SPEED

                    bv.Velocity = dir * speed
                    task.wait()
                end

                bv.Velocity = Vector3.zero
            end

            -- ===== FIND NEAREST =====
            local nearest
            local shortest = math.huge

            for _, obj in pairs(tierFolder:GetChildren()) do
                local part = obj:IsA("Model") and obj:FindFirstChildWhichIsA("BasePart") or obj
                if part and part:IsA("BasePart") then
                    local dist = (root.Position - part.Position).Magnitude
                    if dist < shortest then
                        shortest = dist
                        nearest = part
                    end
                end
            end

            if nearest then
                local startPos = root.Position
                local startXZ = Vector3.new(startPos.X, TARGET_Y, startPos.Z)

                flyTo(startXZ)

                local below = Vector3.new(
                    nearest.Position.X,
                    nearest.Position.Y + BELOW_OFFSET,
                    nearest.Position.Z
                )

                flyTo(below)
                flyTo(nearest.Position)

                local prompt = nearest:FindFirstChildWhichIsA("ProximityPrompt", true)
                if prompt then
                    prompt.HoldDuration = 0
                    fireproximityprompt(prompt)
                end

                task.wait(1)

                flyTo(startXZ)
                flyTo(FINAL_POSITION)
            end

            bv:Destroy()
            bp:Destroy()

            task.wait(0.3)
        end

        -- CLEANUP khi tắt
        if noclipConnection then
            noclipConnection:Disconnect()
            noclipConnection = nil
        end

        farmThread = nil
    end)
end

local function startHighTierFarm()

        while highTierEnabled do

            if not folder then task.wait(1) continue end

            local tierFolder = nil

            for _, tierName in ipairs(priorityTiers) do
                local checkFolder = folder:FindFirstChild(tierName)
                if checkFolder and #checkFolder:GetChildren() > 0 then
                    tierFolder = checkFolder
                    break
                end
            end

            if not tierFolder then
                task.wait(1)
                continue
            end

            -- ===== NOCLIP =====
            local connection
            connection = game:GetService("RunService").Stepped:Connect(function()
                if not highTierEnabled then
                    connection:Disconnect()
                    return
                end

                if player.Character then
                    for _, v in pairs(player.Character:GetDescendants()) do
                        if v:IsA("BasePart") then
                            v.CanCollide = false
                        end
                    end
                end
            end)

            -- ===== BODY MOVERS =====
            local bv = Instance.new("BodyVelocity")
            bv.MaxForce = Vector3.new(1e9,1e9,1e9)
            bv.Parent = root

            local bp = Instance.new("BodyPosition")
            bp.MaxForce = Vector3.new(1e9,1e9,1e9)
            bp.P = 10000
            bp.D = 1000
            bp.Parent = root

            local function flyTo(position)
                bp.Position = position

                while highTierEnabled and root and (root.Position - position).Magnitude > 3 do
                    local distance = (root.Position - position).Magnitude
                    local dir = (position - root.Position).Unit

                    local speed = distance < 40 and 300 or 1000
                    bv.Velocity = dir * speed
                    task.wait()
                end

                bv.Velocity = Vector3.zero
            end

            -- ===== TÌM GẦN NHẤT =====
            local nearest
            local shortest = math.huge

            for _, obj in pairs(tierFolder:GetChildren()) do
                local part = obj:IsA("Model") and obj:FindFirstChildWhichIsA("BasePart") or obj
                if part and part:IsA("BasePart") then
                    local dist = (root.Position - part.Position).Magnitude
                    if dist < shortest then
                        shortest = dist
                        nearest = part
                    end
                end
            end

            if not nearest then
                bv:Destroy()
                bp:Destroy()
                connection:Disconnect()
                task.wait(0.5)
                continue
            end

            -- ===== MOVE LOGIC =====
            local startPos = root.Position
            local startXZ = Vector3.new(startPos.X, -2.74, startPos.Z)

            flyTo(startXZ)

            local below = Vector3.new(
                nearest.Position.X,
                nearest.Position.Y - 6,
                nearest.Position.Z

            flyTo(below)
            flyTo(nearest.Position)

            local prompt = nearest:FindFirstChildWhichIsA("ProximityPrompt", true)
            if prompt then
                prompt.HoldDuration = 0
                fireproximityprompt(prompt)
            end

            task.wait(1)

            flyTo(startXZ)
            flyTo(FINAL_POSITION)

            bv:Destroy()
            bp:Destroy()
            connection:Disconnect()

            task.wait(0.3)
        end
    end)
end
